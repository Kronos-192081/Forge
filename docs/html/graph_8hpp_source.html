158bool Graph<T>::dfsCycleDetection(const T& node, std::unordered_set<T>& visited, std::unordered_set<T>& recStack, std::vector<T>& path, std::vector<T>& cycle) {
159 if (recStack.find(node) != recStack.end()) {
160 auto it = std::find(path.begin(), path.end(), node);
161 cycle.assign(it, path.end());
162 return true;
163 }
164 if (visited.find(node) != visited.end()) {
165 return false;
166 }
167 visited.insert(node);
168 recStack.insert(node);
169 path.push_back(node);

171 if (dfsCycleDetection(neighbor, visited, recStack, path, cycle)) {
172 return true;
173 }
174 }
175 recStack.erase(node);
176 path.pop_back();
177 return false;

181std::vector<T> Graph<T>::topologicalSort() {
182 std::stack<T> Stack;
183 std::unordered_set<T> visited;
184 for (const auto& [node, _] : adjList) {
185 if (visited.find(node) == visited.end()) {
186 dfsTopologicalSort(node, visited, Stack);
187 }
188 }
189 std::vector<T> result;
190 while (!Stack.empty()) {
191 result.push_back(Stack.top());
192 Stack.pop();
193 }
194 return result;
